	\xdef\yyy{\pgfmathresult}
	\pgfmathparse{atan(\yyy/3)}
	\xdef\winkel{\pgfmathresult}
	\draw[color=#2,line width=#3] (3,-\yyy) -- (3,\yyy) arc (\winkel:{360-\winkel}:#1) --cycle;
}
